17  package org.mal_lang.compiler;
18  
19  import java.io.IOException;
20  import java.util.jar.Manifest;
21  import picocli.CommandLine.IVersionProvider;
22  
23  /**
24   * Reads version string from manifest file.
25   *
26   * @since 0.1.0
27   */
28  public class ManifestVersionProvider implements IVersionProvider {
29    @Override
30    public String[] getVersion() throws Exception {
31      try {
32        var resources =
33            ManifestVersionProvider.class.getClassLoader().getResources("META-INF/MANIFEST.MF");
34        while (resources.hasMoreElements()) {
35          var url = resources.nextElement();
36          try (var in = url.openStream()) {
37            var manifest = new Manifest(in);
38            var attributes = manifest.getMainAttributes();
39            var title = attributes.getValue("Implementation-Title");
40            var version = attributes.getValue("Implementation-Version");
41            var vendor = attributes.getValue("Implementation-Vendor");
42            if (title != null
43                && version != null
44                && vendor != null
45                && title.equals("malc")
46                && vendor.equals("MAL")) {
47              return new String[] {String.format("%s version %s", title, version)};
48            }
49          }
50        }
51        throw new RuntimeException("Failed to get resource \"META-INF/MANIFEST.MF\"");
52      } catch (IOException e) {
53        throw new RuntimeException(e);
54      }
55    }
56  }
